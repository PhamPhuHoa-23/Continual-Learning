\section{Setup and Components}
% ═══════════════════════════════════════════════════════════════════════════

\paragraph{Setting.}
Tasks $\mathcal{D}_1,\ldots,\mathcal{D}_T$ arrive sequentially;
$\mathcal{Y}_t \cap \mathcal{Y}_{t'}=\emptyset$; no raw past data stored (exemplar-free).

\paragraph{Components.}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
\textbf{Module} & \textbf{Design} & \textbf{Status across tasks} \\
\midrule
Backbone & ViT pretrained ImageNet-21K & Frozen always \\
Slot Attention & Trained Task~1 with $\Lcal_{\text{recon}}$ & Frozen after Task~1 \\
Agent $a_i$ & MLP: $s_k \to h_k \in \RR^{d_h}$ & Frozen after its task \\
VAE $v_i$ & Encoder/decoder frozen; $(\mu_i,\Sigma_i)$ updated & Weights frozen; stats updated \\
Aggregator & Block-diagonal per-agent keys $w_i$ & Key $w_i$ frozen with agent $i$ \\
Classifier & Analytic SLDA, Welford update & Old class stats never modified \\
\bottomrule
\end{tabular}
\end{center}

\paragraph{Aggregator (dimension-stable).}
Each agent $i$ owns attention key $w_i \in \RR^{d_h}$, frozen when agent $i$ freezes:
\begin{equation}
  H = \sum_k \alpha_k h_k \in \RR^{d_h}, \qquad
  \alpha_k = \frac{\exp(w_{\sigma(k)}^\top h_k)}{\sum_{k'}\exp(w_{\sigma(k')}^\top h_{k'})}.
\end{equation}
Output dimension $d_h$ is \textbf{constant} regardless of agent count.

\begin{proposition}[Zero forgetting]
If agents $\{a_i\}_{i\leq M_{\mathrm{old}}}$ and keys $\{w_i\}_{i\leq M_{\mathrm{old}}}$ are frozen,
then for any old-class image whose slots all route to frozen agents,
$H$ is identical across all future tasks, so SLDA discriminants for old classes never change.
\end{proposition}

\section{Loss Functions}
% ═══════════════════════════════════════════════════════════════════════════

\paragraph{$\Lcal_{\mathrm{recon}}$ — Slot reconstruction (Slot Attention only).}
\begin{equation}
  \Lcal_{\mathrm{recon}} = \|E - \tilde{E}\|^2.
\end{equation}

\paragraph{$\Lcal_p$ — Primitive loss (concept-level, matrix-level).}
For batch with aggregated reps $\{H_i\}$ and one-hot labels $\{y_i\}$:
\begin{equation}
  d^y_{ij} = \frac{\mathbf{1}[y_i=y_j]}{\sum_k \mathbf{1}[y_i=y_k]}, \quad
  d^H_{ij} = \frac{\exp(\tau\,\mathrm{sim}(H_i,H_j))}{\sum_k\exp(\tau\,\mathrm{sim}(H_i,H_k))}, \quad
  \Lcal_p = \sum_{i,j} d^y_{ij}\log\frac{d^y_{ij}}{d^H_{ij}}.
\end{equation}
KL divergence between label and hidden-label similarity matrices.
Uses only image-level labels; no slot-level supervision required.

\paragraph{$\Lcal_{\mathrm{SupCon}}$ — Supervised contrastive loss (pair-level metric).}
\begin{equation}
  \Lcal_{\mathrm{SupCon}} = \sum_i \frac{-1}{|P(i)|}\sum_{p\in P(i)}
  \log\frac{\exp(\mathrm{sim}(H_i,H_p)/\tau)}{\sum_{a\neq i}\exp(\mathrm{sim}(H_i,H_a)/\tau)},
\end{equation}
where $P(i)$ is the set of same-class images in the batch.
Enforces within-class compactness and between-class separability directly on pairs,
complementing $\Lcal_p$ which operates at the matrix level.

\paragraph{$\Lcal_{\mathrm{agent}}$ — Agent reconstruction (anti-collapse).}
Each agent $a_i$ has a lightweight decoder $a_i^{\mathrm{dec}}$:
\begin{equation}
  \Lcal_{\mathrm{agent}} = \frac{1}{K}\sum_k \|a^{\mathrm{dec}}_{\sigma(k)}(h_k) - s_k\|^2.
\end{equation}
Prevents agents from collapsing to trivial identity mappings.

\paragraph{$\Lcal_{\mathrm{local}}$ — Neighbor-robustness loss (local geometry).}
From Gao et al.~\cite{gao2023geometry}, adapted to hidden-label space $H$:
\begin{align}
  e^{t-1}_{ij} &= w^{t-1}_{ij} - b^{t-1}_{ij},
  \quad
  w^{t-1}_{ij} = \mathbf{1}[\text{same class, close at }t\!-\!1],\;
  b^{t-1}_{ij} = \mathbf{1}[\text{diff class, close at }t\!-\!1], \\
  \Lcal_{\mathrm{local}} &= \sum_{(i,j)} e^{t-1}_{ij}\cdot\|H_i^t - H_j^t\|^2.
\end{align}
Enforces within-class compactness and between-class separability in local neighborhoods,
making $H$ well-structured for SLDA's Gaussian assumption.
Requires a small buffer of exemplars to compute $e^{t-1}_{ij}$ (neighbor affinities from step $t-1$).

\paragraph{Total loss.}
\begin{equation}
\boxed{
  \Lcal = \underbrace{\alpha\,\Lcal_p}_{\text{matrix-level}} +
          \underbrace{\beta\,\Lcal_{\mathrm{SupCon}}}_{\text{pair-level}} +
          \underbrace{\gamma\,\Lcal_{\mathrm{agent}}}_{\text{anti-collapse}} +
          \underbrace{\delta\,\Lcal_{\mathrm{local}}}_{\text{local geometry}}
}
\end{equation}

\section{Routing and Novelty Detection}
% ═══════════════════════════════════════════════════════════════════════════

\paragraph{VAE routing score.}
We use Mahalanobis distance in the frozen VAE latent space
(computationally cheaper than full ELBO, avoids dimension mismatch between $s$ and $z$):
\begin{equation}
  \text{score}(s,i) = -\bigl(\mu_{q_i}(s) - \mu_i\bigr)^\top \Sigma_i^{-1}
                       \bigl(\mu_{q_i}(s) - \mu_i\bigr),
\end{equation}
where $\mu_{q_i}(s)$ is the VAE posterior mean (frozen encoder),
and $(\mu_i,\Sigma_i)$ are the \textbf{aggregate latent statistics} of agent $i$,
updated incrementally via Welford's algorithm whenever new slots are routed to agent $i$.
This separates the VAE \emph{network} (frozen) from the VAE \emph{statistics} (updated),
allowing routing to adapt to new variations of a sub-concept without forgetting.

\paragraph{Routing decision.}
\begin{equation}
  \sigma(k) = \begin{cases}
    \arg\max_i\;\mathrm{score}(s_k,i) & \text{if } \max_i\;\mathrm{score}(s_k,i)\geq\theta_{\mathrm{match}}, \\
    \texttt{unassigned} & \text{if } \max_i\;\mathrm{score}(s_k,i)<\theta_{\mathrm{novel}}.
  \end{cases}
\end{equation}

\paragraph{Agent spawning.}
Unassigned slots collected into buffer $\mathcal{B}$.
When $|\mathcal{B}|\geq B_{\min}$: cluster $\mathcal{B}$;
spawn one new agent per cluster with size $\geq n_{\min}$ and
intra-cluster cosine similarity $\geq \rho_{\min}$.

\section{Training Protocol}
% ═══════════════════════════════════════════════════════════════════════════

\begin{algorithm}[H]
\caption{Full Training Protocol}
\begin{algorithmic}[1]
\Require Task stream $\mathcal{D}_1,\ldots,\mathcal{D}_T$, frozen ViT backbone
\Ensure Agent pool $\{a_i\}$, VAE pool $\{v_i\}$, keys $\{w_i\}$, SLDA

\State \textbf{// --- Task 1 ---}
\State Train Slot Attention with $\Lcal_{\mathrm{recon}}$ on $\mathcal{D}_1$; \textbf{Freeze} $\theta_s$ \Comment{Warm-up}
\State $\{C_j\}_{j=1}^{M_0}\leftarrow\mathrm{KMeans}(\text{all slots of }\mathcal{D}_1)$
\State Spawn $M_0$ agents $\{a_i\}$ and keys $\{w_i\}$
\State Train $\{a_i\},\{w_i\}$ with $\Lcal=\alpha\Lcal_p+\beta\Lcal_{\mathrm{SupCon}}+\gamma\Lcal_{\mathrm{agent}}$
\For{each agent $a_i$}
  \State Collect routed slots $\to$ Train VAE $v_i$ $\to$ \textbf{Freeze} $v_i$ network
  \State Init latent stats $(\mu_i,\Sigma_i)$ from routed slots via Welford
\EndFor
\State \textbf{Freeze} all $a_i$, all $w_i$
\State Update SLDA with $\mathcal{D}_1$

\State \textbf{// --- Tasks $t=2,\ldots,T$ ---}
\For{$t=2$ to $T$}
  \State $\mathcal{B}\leftarrow\emptyset$
  \For{each image $x\in\mathcal{D}_t$}
    \State $\{s_k\}\leftarrow\mathrm{SlotAttn}(x)$
    \For{each slot $s_k$}
      \State $i^*\leftarrow\arg\max_i\;\mathrm{score}(s_k,i)$
      \If{$\mathrm{score}(s_k,i^*)\geq\theta_{\mathrm{match}}$}
        \State Assign $s_k\to a_{i^*}$; update $(\mu_{i^*},\Sigma_{i^*})$ via Welford \Comment{VAE stats only}
      \Else
        \State $\mathcal{B}\leftarrow\mathcal{B}\cup\{s_k\}$
      \EndIf
    \EndFor
  \EndFor
  \If{$|\mathcal{B}|\geq B_{\min}$}
    \State Cluster $\mathcal{B}$; spawn $J^*$ new agents $\{a_{\mathrm{new}}\}$, keys $\{w_{\mathrm{new}}\}$
  \EndIf
  \State Train $\{a_{\mathrm{new}}\},\{w_{\mathrm{new}}\}$ with
         $\Lcal=\alpha\Lcal_p+\beta\Lcal_{\mathrm{SupCon}}+\gamma\Lcal_{\mathrm{agent}}+\delta\Lcal_{\mathrm{local}}$
  \For{each new agent $a_j$}
    \State Train VAE $v_j$ $\to$ \textbf{Freeze} $v_j$ network; init $(\mu_j,\Sigma_j)$
  \EndFor
  \State \textbf{Freeze} $\{a_{\mathrm{new}}\}$, $\{w_{\mathrm{new}}\}$
  \State Update SLDA with $\mathcal{D}_t$ (new classes only; old stats unchanged)
\EndFor
\end{algorithmic}
\end{algorithm}

\section{Design Decisions Summary}
% ═══════════════════════════════════════════════════════════════════════════

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{p{4cm}p{5cm}p{5cm}}
\toprule
\textbf{Decision} & \textbf{Choice} & \textbf{Rationale} \\
\midrule
VAE network vs.\ stats & Network frozen; $(\mu_i,\Sigma_i)$ updated & Avoids forgetting in VAE; routing adapts to new variations of sub-concepts \\
Routing score & Mahalanobis in latent space & Probabilistically grounded; no dimension mismatch between $s$ and $z$ \\
Aggregator & Block-diagonal per-agent keys & Constant output $d_h$; old-class $H$ never drifts \\
$H$ structure & $\Lcal_p + \Lcal_{\mathrm{SupCon}} + \Lcal_{\mathrm{local}}$ & Matrix-level + pair-level + neighborhood-level enforce Gaussian-like clusters for SLDA \\
Agent collapse & $\Lcal_{\mathrm{agent}}$ reconstruction & Prevents trivial slot-to-embedding mappings \\
Exemplar buffer & Small buffer for $\Lcal_{\mathrm{local}}$ only & Not for replay; only to compute neighbor affinities $e^{t-1}_{ij}$ \\
\bottomrule
\end{tabular}
\end{center}

\paragraph{Open question.}
$\Lcal_{\mathrm{local}}$ requires neighbor affinities $e^{t-1}_{ij}$ from old data.
A small buffer ($\sim$20 exemplars/class) suffices to compute these affinities;
this is a weaker requirement than full replay.
If truly exemplar-free is required, $\Lcal_{\mathrm{local}}$ can be dropped
at a likely cost to SLDA classification quality.
